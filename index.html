import numpy as np import matplotlib.pyplot as plt a = [] b=[] xdata1 = float("x1") ydata1 = float("y1") xdata2 = float("x2") ydata2 = float("y2") a.append(xdata1) b.append(ydata1) a.append(xdata2) b.append(ydata2) a = np.array(a) b = np.array(b) afloat = [float(i) for i in a] bfloat = [float(i) for i in b] # Generate model function x_model = np.arange(0.0,50.0,.01) y_model = 2*x_model*x_model fig1,ax1 = plt.subplots(1,dpi=150,figsize=(5,3)) fig1, ax1 = plt.subplots() ax1.scatter(afloat,bfloat) ax1.set_title("Graph of Data",fontsize=11) ax1.set_xlabel("x variable",fontsize=10) ax1.set_ylabel("y variable",fontsize=10) plt.plot(x_model, y_model) ax1.set_xlim(0,6) ax1.set_ylim(0,50) ax1.margins(y=0) ax1.grid() fig1
